// EventMonitor.cs
using Weds.Core.Devices;
using Microsoft.Extensions.Logging;

public class EventMonitor
{
    private readonly ILogger _logger;
    
    public EventMonitor()
    {
        var loggerFactory = LoggerFactory.Create(builder => builder.AddConsole());
        _logger = loggerFactory.CreateLogger<EventMonitor>();
    }
    
    public async Task MonitorDeviceAsync()
    {
        var mockDevice = new MockDevice(_logger.CreateLogger<MockDevice>());
        
        // 訂閱事件
        mockDevice.StatusChanged += OnStatusChanged;
        mockDevice.TelemetryDataAvailable += OnTelemetryData;
        mockDevice.DeviceEventOccurred += OnDeviceEvent;
        
        // 初始化和啟動裝置
        var config = CreateMockConfiguration();
        await mockDevice.InitializeAsync(config);
        await mockDevice.StartAsync();
        
        // 監控 30 秒
        await Task.Delay(30000);
        
        await mockDevice.StopAsync();
        mockDevice.Dispose();
    }
    
    private void OnStatusChanged(object sender, DeviceStatusChangedEventArgs e)
    {
        _logger.LogInformation("Status Changed: {OldStatus} -> {NewStatus}", 
            e.OldStatus, e.NewStatus);
    }
    
    private void OnTelemetryData(object sender, TelemetryDataEventArgs e)
    {
        _logger.LogInformation("Telemetry Data from {DeviceId}: {DataCount} values", 
            e.DeviceId, e.SensorData.Values.Count);
    }
    
    private void OnDeviceEvent(object sender, DeviceEventArgs e)
    {
        _logger.LogInformation("Device Event: {EventType} - {Message}", 
            e.EventType, e.Message);
    }
}

// SerialTest.cs
using System.IO.Ports;
using System.Text;

public class SerialTest
{
    public async Task TestSerialCommunication()
    {
        // 設定串列埠（依照你的虛擬埠設定調整）
        var masterPort = new SerialPort("COM3", 9600, Parity.None, 8, StopBits.One);
        var slavePort = new SerialPort("COM4", 9600, Parity.None, 8, StopBits.One);
        
        try
        {
            masterPort.Open();
            slavePort.Open();
            
            // 從站接收處理
            slavePort.DataReceived += (sender, e) =>
            {
                var port = sender as SerialPort;
                var data = port.ReadExisting();
                Console.WriteLine($"Slave received: {data}");
                
                // Echo back
                port.Write($"Echo: {data}");
            };
            
            // 主站發送測試
            for (int i = 0; i < 5; i++)
            {
                var message = $"Test message {i}";
                masterPort.WriteLine(message);
                Console.WriteLine($"Master sent: {message}");
                
                await Task.Delay(1000);
            }
            
            await Task.Delay(2000); // 等待回應
        }
        finally
        {
            masterPort?.Close();
            slavePort?.Close();
        }
    }
}

練習 6：實作自訂 Serial Communication#

基於 ICommunication 介面實作：

// CustomSerialCommunication.cs
using System.IO.Ports;
using Weds.Abstractions;
using Weds.Abstractions.Models;
using Weds.Abstractions.Events;

public class CustomSerialCommunication : ICommunication
{
    private SerialPort _serialPort;
    private bool _disposed = false;
    
    public string CommunicationType => "Custom Serial";
    public bool IsConnected => _serialPort?.IsOpen ?? false;
    
    public event EventHandler<CommunicationEventArgs> DataReceived;
    public event EventHandler<CommunicationEventArgs> EventReceived;
    public event EventHandler<CommunicationStatusEventArgs> ConnectionStatusChanged;
    
    public async Task<bool> InitializeAsync(CommunicationConfiguration config)
    {
        try
        {
            var portName = config.ConnectionString;
            var baudRate = config.Parameters.GetValueOrDefault("BaudRate", "9600");
            
            _serialPort = new SerialPort(portName, int.Parse(baudRate));
            _serialPort.DataReceived += OnSerialDataReceived;
            
            return true;
        }
        catch (Exception ex)
        {
            Console.WriteLine($"Serial initialization error: {ex.Message}");
            return false;
        }
    }
    
    public async Task<T> ReadAsync<T>(string address, CancellationToken cancellationToken = default)
    {
        if (!IsConnected) 
            throw new InvalidOperationException("Serial port not connected");
            
        // 實作讀取邏輯
        // address 可以是命令格式，例如："READ_COIL:1" 或 "READ_HOLDING:0x100"
        
        var command = BuildReadCommand(address);
        await WriteRawDataAsync(command);
        
        // 等待回應並解析
        var response = await WaitForResponseAsync(cancellationToken);
        return ParseResponse<T>(response);
    }
    
    public async Task<bool> WriteAsync<T>(string address, T value, CancellationToken cancellationToken = default)
    {
        if (!IsConnected) 
            return false;
            
        var command = BuildWriteCommand(address, value);
        await WriteRawDataAsync(command);
        return true;
    }
    
    private void OnSerialDataReceived(object sender, SerialDataReceivedEventArgs e)
    {
        var port = sender as SerialPort;
        var data = port.ReadExisting();
        
        DataReceived?.Invoke(this, new CommunicationEventArgs(
            CommunicationType, 
            data, 
            DateTime.UtcNow
        ));
    }
    
    // 其他必要方法的實作...
}

// ModbusTcpTest.cs
using System.Net.Sockets;

public class ModbusTcpTest
{
    public async Task TestModbusTcp()
    {
        using var client = new TcpClient();
        await client.ConnectAsync("127.0.0.1", 502);
        
        using var stream = client.GetStream();
        
        // 讀取保持暫存器 (功能碼 0x03)
        var readHoldingRequest = new byte[] {
            0x00, 0x01, // Transaction ID
            0x00, 0x00, // Protocol ID
            0x00, 0x06, // Length
            0x01,       // Unit ID
            0x03,       // Function Code (Read Holding Registers)
            0x00, 0x00, // Starting Address
            0x00, 0x02  // Quantity
        };
        
        await stream.WriteAsync(readHoldingRequest);
        
        var buffer = new byte[256];
        var bytesRead = await stream.ReadAsync(buffer);
        
        Console.WriteLine("Modbus TCP Response:");
        Console.WriteLine(BitConverter.ToString(buffer, 0, bytesRead));
        
        // 解析回應
        if (bytesRead >= 9 && buffer[7] == 0x03)
        {
            var dataLength = buffer[8];
            Console.WriteLine($"Data length: {dataLength} bytes");
            
            for (int i = 0; i < dataLength / 2; i++)
            {
                var value = (buffer[9 + i * 2] << 8) | buffer[10 + i * 2];
                Console.WriteLine($"Register {i}: {value}");
            }
        }
    }
}

練習 9：Modbus RTU 模擬#

使用虛擬串列埠實作 Modbus RTU：

// ModbusRtuTest.cs
using System.IO.Ports;

public class ModbusRtuTest
{
    public async Task TestModbusRtu()
    {
        // 模擬從站
        var slaveTask = Task.Run(() => ModbusRtuSlave("COM4"));
        
        // 主站測試
        using var masterPort = new SerialPort("COM3", 9600, Parity.None, 8, StopBits.One);
        masterPort.Open();
        
        // 讀取保持暫存器命令 (從站 ID: 1, 功能碼: 3, 地址: 0, 數量: 2)
        var command = new byte[] { 0x01, 0x03, 0x00, 0x00, 0x00, 0x02 };
        var crc = CalculateCrc16(command);
        var fullCommand = command.Concat(BitConverter.GetBytes(crc)).ToArray();
        
        masterPort.Write(fullCommand, 0, fullCommand.Length);
        
        await Task.Delay(100); // 等待回應
        
        if (masterPort.BytesToRead > 0)
        {
            var buffer = new byte[masterPort.BytesToRead];
            masterPort.Read(buffer, 0, buffer.Length);
            
            Console.WriteLine("Modbus RTU Response:");
            Console.WriteLine(BitConverter.ToString(buffer));
        }
        
        masterPort.Close();
    }
    
    private void ModbusRtuSlave(string portName)
    {
        using var slavePort = new SerialPort(portName, 9600, Parity.None, 8, StopBits.One);
        slavePort.Open();
        
        var buffer = new byte[256];
        
        while (true)
        {
            if (slavePort.BytesToRead > 0)
            {
                var bytesRead = slavePort.Read(buffer, 0, buffer.Length);
                
                // 驗證 CRC 和處理命令
                if (ValidateCrc(buffer, bytesRead))
                {
                    var response = ProcessModbusCommand(buffer, bytesRead);
                    if (response != null)
                    {
                        slavePort.Write(response, 0, response.Length);
                    }
                }
            }
            
            Thread.Sleep(10);
        }
    }
    
    private byte[] ProcessModbusCommand(byte[] command, int length)
    {
        if (length < 8) return null; // 最小 Modbus 封包大小
        
        var slaveId = command[0];
        var functionCode = command[1];
        
        if (slaveId != 1) return null; // 只處理從站 ID 1
        
        switch (functionCode)
        {
            case 0x03: // 讀取保持暫存器
                return ProcessReadHoldingRegisters(command);
            default:
                return CreateExceptionResponse(slaveId, functionCode, 0x01); // 非法功能碼
        }
    }
    
    private byte[] ProcessReadHoldingRegisters(byte[] command)
    {
        var startAddress = (command[2] << 8) | command[3];
        var quantity = (command[4] << 8) | command[5];
        
        // 模擬暫存器值
        var registerValues = new ushort[] { 1234, 5678, 9012, 3456 };
        
        var response = new List<byte>
        {
            0x01, // 從站 ID
            0x03, // 功能碼
            (byte)(quantity * 2) // 資料位元組數
        };
        
        for (int i = 0; i < quantity && (startAddress + i) < registerValues.Length; i++)
        {
            var value = registerValues[startAddress + i];
            response.Add((byte)(value >> 8));   // 高位元組
            response.Add((byte)(value & 0xFF)); // 低位元組
        }
        
        // 計算並添加 CRC
        var crc = CalculateCrc16(response.ToArray());
        response.AddRange(BitConverter.GetBytes(crc));
        
        return response.ToArray();
    }
}

練習 10：HTTP 裝置模擬器#

建立簡單的 HTTP 裝置模擬器：

// HttpDeviceSimulator.cs
using Microsoft.AspNetCore.Builder;
using Microsoft.AspNetCore.Http;
using System.Text.Json;

public class HttpDeviceSimulator
{
    private readonly Dictionary<int, bool> _digitalInputs = new();
    private readonly Dictionary<int, bool> _digitalOutputs = new();
    
    public async Task StartAsync()
    {
        // 初始化模擬資料
        for (int i = 0; i < 8; i++)
        {
            _digitalInputs[i] = Random.Shared.Next(0, 2) == 1;
            _digitalOutputs[i] = false;
        }
        
        var builder = WebApplication.CreateBuilder();
        var app = builder.Build();
        
        // GET /digitalinput/all/value
        app.MapGet("/digitalinput/all/value", () =>
        {
            return Results.Ok(_digitalInputs.ToDictionary(
                kvp => kvp.Key.ToString(),
                kvp => kvp.Value
            ));
        });
        
        // GET /digitalinput/{channel}/value
        app.MapGet("/digitalinput/{channel}/value", (int channel) =>
        {
            if (_digitalInputs.ContainsKey(channel))
            {
                return Results.Ok(new Dictionary<string, bool> 
                { 
                    { channel.ToString(), _digitalInputs[channel] } 
                });
            }
            return Results.NotFound();
        });
        
        // POST /digitaloutput/{channel}/value
        app.MapPost("/digitaloutput/{channel}/value", async (int channel, HttpContext context) =>
        {
            using var reader = new StreamReader(context.Request.Body);
            var json = await reader.ReadToEndAsync();
            var data = JsonSerializer.Deserialize<Dictionary<string, object>>(json);
            
            if (data.ContainsKey("Value"))
            {
                var value = Convert.ToBoolean(data["Value"]);
                _digitalOutputs[channel] = value;
                
                Console.WriteLine($"Digital Output {channel} set to {(value ? "HIGH" : "LOW")}");
                return Results.Ok(new { success = true });
            }
            
            return Results.BadRequest();
        });
        
        // GET /digitaloutput/all/value
        app.MapGet("/digitaloutput/all/value", () =>
        {
            return Results.Ok(_digitalOutputs.ToDictionary(
                kvp => kvp.Key.ToString(),
                kvp => kvp.Value
            ));
        });
        
        Console.WriteLine("HTTP Device Simulator started on http://localhost:8080");
        await app.RunAsync("http://localhost:8080");
    }
}

// TemperatureHumiditySensor.cs
using Microsoft.Extensions.Logging;
using Weds.Abstractions;
using Weds.Abstractions.Models;
using Weds.Abstractions.Events;
using Weds.Abstractions.Enums;

public class TemperatureHumiditySensor : IDevice
{
    private readonly ILogger<TemperatureHumiditySensor> _logger;
    private System.Timers.Timer _pollingTimer;
    private CancellationTokenSource _cancellationTokenSource;
    private readonly Random _random = new();
    
    // 模擬感測器資料
    private double _temperature = 25.0;
    private double _humidity = 50.0;
    private double _pressure = 1013.25;
    
    public string DeviceId { get; private set; } = "TH-SENSOR-001";
    public string DeviceType => "Temperature Humidity Sensor";
    public DeviceConfiguration Configuration { get; private set; }
    public DeviceStatus Status { get; private set; } = DeviceStatus.Uninitialized;
    
    // 事件定義
    public event EventHandler<DeviceStatusChangedEventArgs> StatusChanged;
    public event EventHandler<TelemetryDataEventArgs> TelemetryDataAvailable;
    public event EventHandler<DeviceHealthEventArgs> HealthStatusChanged;
    public event EventHandler<DeviceEventArgs> DeviceEventOccurred;
    
    public TemperatureHumiditySensor(ILogger<TemperatureHumiditySensor> logger)
    {
        _logger = logger ?? throw new ArgumentNullException(nameof(logger));
    }
    
    public Task<bool> InitializeAsync(DeviceConfiguration configuration)
    {
        try
        {
            _logger.LogInformation("Initializing Temperature Humidity Sensor {DeviceId}", DeviceId);
            
            Configuration = configuration ?? throw new ArgumentNullException(nameof(configuration));
            
            // 提取配置
            if (configuration.ConnectionSettings.TryGetValue("DeviceId", out var deviceId))
            {
                DeviceId = deviceId.ToString();
            }
            
            if (configuration.ConnectionSettings.TryGetValue("InitialTemperature", out var temp))
            {
                _temperature = Convert.ToDouble(temp);
            }
            
            if (configuration.ConnectionSettings.TryGetValue("InitialHumidity", out var hum))
            {
                _humidity = Convert.ToDouble(hum);
            }
            
            UpdateStatus(DeviceStatus.Initialized);
            _logger.LogInformation("Sensor {DeviceId} initialized successfully", DeviceId);
            return Task.FromResult(true);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to initialize sensor {DeviceId}", DeviceId);
            UpdateStatus(DeviceStatus.Error);
            return Task.FromResult(false);
        }
    }
    
    public Task<bool> StartAsync(CancellationToken cancellationToken = default)
    {
        try
        {
            _logger.LogInformation("Starting sensor {DeviceId}", DeviceId);
            
            _cancellationTokenSource = new CancellationTokenSource();
            UpdateStatus(DeviceStatus.Running);
            
            // 啟動資料採集計時器
            _pollingTimer = new System.Timers.Timer(2000); // 每 2 秒
            _pollingTimer.Elapsed += async (sender, e) => await CollectSensorDataAsync();
            _pollingTimer.Start();
            
            _logger.LogInformation("Sensor {DeviceId} started successfully", DeviceId);
            return Task.FromResult(true);
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to start sensor {DeviceId}", DeviceId);
            UpdateStatus(DeviceStatus.Error);
            return Task.FromResult(false);
        }
    }
    
    public Task<CommandResult> ExecuteCommandAsync(DeviceCommand command)
    {
        try
        {
            _logger.LogInformation("Executing command {CommandName} on sensor {DeviceId}", 
                command.CommandName, DeviceId);
            
            switch (command.CommandName.ToLower())
            {
                case "calibrate":
                    return CalibrateAsync(command);
                    
                case "setthreshold":
                    return SetThresholdAsync(command);
                    
                case "getreadings":
                    return GetReadingsAsync();
                    
                default:
                    return Task.FromResult(new CommandResult
                    {
                        Success = false,
                        Message = $"Unknown command: {command.CommandName}"
                    });
            }
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Failed to execute command {CommandName}", command.CommandName);
            return Task.FromResult(new CommandResult
            {
                Success = false,
                Message = ex.Message
            });
        }
    }
    
    private async Task CollectSensorDataAsync()
    {
        if (_cancellationTokenSource?.Token.IsCancellationRequested ?? true)
            return;
        
        try
        {
            // 模擬感測器讀數變化
            _temperature += (_random.NextDouble() - 0.5) * 2; // ±1°C 變化
            _humidity += (_random.NextDouble() - 0.5) * 4;    // ±2% 變化
            _pressure += (_random.NextDouble() - 0.5) * 2;    // ±1 hPa 變化
            
            // 限制範圍
            _temperature = Math.Clamp(_temperature, -40, 85);
            _humidity = Math.Clamp(_humidity, 0, 100);
            _pressure = Math.Clamp(_pressure, 300, 1100);
            
            // 檢查警報條件
            CheckAlarmConditions();
            
            // 發送遙測資料
            var telemetryData = new Dictionary<string, object>
            {
                { "Temperature", Math.Round(_temperature, 2) },
                { "Humidity", Math.Round(_humidity, 2) },
                { "Pressure", Math.Round(_pressure, 2) },
                { "DeviceId", DeviceId },
                { "Timestamp", DateTime.UtcNow.ToString("yyyy-MM-ddTHH:mm:ssZ") }
            };
            
            var sensorData = new SensorDataPacket
            {
                SensorId = DeviceId,
                SensorType = DeviceType,
                Values = telemetryData,
                Timestamp = DateTime.UtcNow
            };
            
            TelemetryDataAvailable?.Invoke(this, new TelemetryDataEventArgs(DeviceId, sensorData));
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "Error collecting sensor data for {DeviceId}", DeviceId);
        }
    }
    
    private void CheckAlarmConditions()
    {
        // 溫度過高警報
        if (_temperature > 40)
        {
            DeviceEventOccurred?.Invoke(this, new DeviceEventArgs(
                DeviceId,
                DeviceEventSeverity.Warning,
                "HighTemperature",
                $"Temperature too high: {_temperature:F2}°C",
                new Dictionary<string, object> { { "Temperature", _temperature } }
            ));
        }
        
        // 濕度過高警報
        if (_humidity > 90)
        {
            DeviceEventOccurred?.Invoke(this, new DeviceEventArgs(
                DeviceId,
                DeviceEventSeverity.Warning,
                "HighHumidity",
                $"Humidity too high: {_humidity:F2}%",
                new Dictionary<string, object> { { "Humidity", _humidity } }
            ));
        }
    }
    
    // 其他必要方法實作...
}

        await mockDevice.StopAsync();
    }

        {

            Parameters = new Dictionary<string, object> { { "index", index } }
        };
        
        var result = await device.ExecuteCommandAsync(command);

// MultiProtocolGateway.cs
public class MultiProtocolGateway
{
    private readonly Dictionary<string, IDevice> _devices = new();
    private readonly Dictionary<string, ICommunication> _communications = new();
    
    public async Task AddDeviceAsync(string deviceId, IDevice device, ICommunication communication)
    {
        _devices[deviceId] = device;
        _communications[deviceId] = communication;
        
        // 訂閱裝置事件
        device.TelemetryDataAvailable += OnTelemetryData;
        device.StatusChanged += OnDeviceStatusChanged;
    }
    
    public async Task<object> ReadFromDeviceAsync(string deviceId, string address)
    {
        if (_communications.ContainsKey(deviceId))
        {
            return await _communications[deviceId].ReadAsync<object>(address);
        }
        throw new ArgumentException($"Device {deviceId} not found");
    }
    
    public async Task<bool> WriteToDeviceAsync(string deviceId, string address, object value)
    {
        if (_communications.ContainsKey(deviceId))
        {
            return await _communications[deviceId].WriteAsync(address, value);
        }
        return false;
    }
    
    private void OnTelemetryData(object sender, TelemetryDataEventArgs e)
    {
        // 轉發到雲端或其他系統
        Console.WriteLine($"Gateway received telemetry from {e.DeviceId}: {e.SensorData.Values.Count} values");
    }
    
    private void OnDeviceStatusChanged(object sender, DeviceStatusChangedEventArgs e)
    {
        Console.WriteLine($"Gateway: Device {e.DeviceId} status changed to {e.NewStatus}");
    }
}

// DataCollectionSystem.cs
public class DataCollectionSystem
{
    private readonly List<IDevice> _devices = new();
    private readonly Queue<SensorDataPacket> _dataQueue = new();
    private readonly Timer _analysisTimer;
    
    public DataCollectionSystem()
    {
        _analysisTimer = new Timer(AnalyzeData, null, TimeSpan.Zero, TimeSpan.FromMinutes(1));
    }
    
    public void AddDevice(IDevice device)
    {
        _devices.Add(device);
        device.TelemetryDataAvailable += OnTelemetryDataReceived;
    }
    
    private void OnTelemetryDataReceived(object sender, TelemetryDataEventArgs e)
    {
        lock (_dataQueue)
        {
            _dataQueue.Enqueue(e.SensorData);
            
            // 保持隊列大小
            while (_dataQueue.Count > 1000)
            {
                _dataQueue.Dequeue();
            }
        }
    }
    
    private void AnalyzeData(object state)
    {
        List<SensorDataPacket> dataToAnalyze;
        
        lock (_dataQueue)
        {
            dataToAnalyze = new List<SensorDataPacket>(_dataQueue);
            _dataQueue.Clear();
        }
        
        if (dataToAnalyze.Count == 0) return;
        
        // 進行資料分析
        var temperatureData = dataToAnalyze
            .Where(d => d.Values.ContainsKey("Temperature"))
            .Select(d => Convert.ToDouble(d.Values["Temperature"]))
            .ToList();
        
        if (temperatureData.Count > 0)
        {
            var avgTemp = temperatureData.Average();
            var minTemp = temperatureData.Min();
            var maxTemp = temperatureData.Max();
            
            Console.WriteLine($"分析結果 - 溫度：平均 {avgTemp:F2}°C, 最低 {minTemp:F2}°C, 最高 {maxTemp:F2}°C");
        }
        
        // 可以添加更多分析邏輯，如異常檢測、趨勢分析等
    }
}

// WebDashboard.cs
using Microsoft.AspNetCore.Builder;
using Microsoft.AspNetCore.Http;
using System.Text.Json;

public class WebDashboard
{
    private readonly Dictionary<string, IDevice> _devices = new();
    private readonly Dictionary<string, object> _latestData = new();
    
    public void AddDevice(string deviceId, IDevice device)
    {
        _devices[deviceId] = device;
        device.TelemetryDataAvailable += (sender, e) =>
        {
            _latestData[deviceId] = e.SensorData.Values;
        };
    }
    
    public async Task StartAsync()
    {
        var builder = WebApplication.CreateBuilder();
        var app = builder.Build();
        
        // 靜態檔案服務
        app.UseStaticFiles();
        
        // API 端點
        app.MapGet("/api/devices", () =>
        {
            return _devices.Keys.ToList();
        });
        
        app.MapGet("/api/devices/{deviceId}/data", (string deviceId) =>
        {
            if (_latestData.ContainsKey(deviceId))
            {
                return Results.Ok(_latestData[deviceId]);
            }
            return Results.NotFound();
        });
        
        app.MapGet("/api/devices/{deviceId}/status", async (string deviceId) =>
        {
            if (_devices.ContainsKey(deviceId))
            {
                var device = _devices[deviceId];
                var health = await device.GetHealthAsync();
                return Results.Ok(new
                {
                    DeviceId = deviceId,
                    Status = device.Status.ToString(),
                    Health = health
                });
            }
            return Results.NotFound();
        });
        
        // 主頁
        app.MapGet("/", async context =>
        {
            var html = GenerateDashboardHtml();
            context.Response.ContentType = "text/html";
            await context.Response.WriteAsync(html);
        });
        
        Console.WriteLine("Web Dashboard started on http://localhost:5000");
        await app.RunAsync("http://localhost:5000");
    }
    
    private string GenerateDashboardHtml()
    {
        return @"
<!DOCTYPE html>
<html>
<head>
    <title>WEDS Device Dashboard</title>
    <script src='https://cdn.jsdelivr.net/npm/chart.js'></script>
    <style>
        body { font-family: Arial, sans-serif; margin: 20px; }
        .device-card { border: 1px solid #ccc; margin: 10px; padding: 15px; border-radius: 5px; }
        .status-ok { color: green; }
        .status-error { color: red; }
    </style>
</head>
<body>
    <h1>WEDS Device Dashboard</h1>
    <div id='devices'></div>
    
    <script>
        async function loadDevices() {
            const response = await fetch('/api/devices');
            const devices = await response.json();
            
            const container = document.getElementById('devices');
            container.innerHTML = '';
            
            for (const deviceId of devices) {
                const deviceDiv = document.createElement('div');
                deviceDiv.className = 'device-card';
                deviceDiv.innerHTML = `<h3>${deviceId}</h3><div id='${deviceId}-status'>Loading...</div>`;
                container.appendChild(deviceDiv);
                
                // 載入裝置狀態
                loadDeviceStatus(deviceId);
            }
        }
        
        async function loadDeviceStatus(deviceId) {
            try {
                const [statusResponse, dataResponse] = await Promise.all([
                    fetch(`/api/devices/${deviceId}/status`),
                    fetch(`/api/devices/${deviceId}/data`)
                ]);
                
                const status = await statusResponse.json();
                const data = dataResponse.ok ? await dataResponse.json() : {};
                
                const statusDiv = document.getElementById(`${deviceId}-status`);
                statusDiv.innerHTML = `
                    <p>Status: <span class='status-${status.Health.IsHealthy ? 'ok' : 'error'}'>${status.Status}</span></p>
                    <p>Health: ${status.Health.Message}</p>
                    <pre>${JSON.stringify(data, null, 2)}</pre>
                `;
            } catch (error) {
                console.error('Error loading device status:', error);
            }
        }
        
        // 每 5 秒更新一次
        setInterval(loadDevices, 5000);
        loadDevices();
    </script>
</body>
</html>";
    }
