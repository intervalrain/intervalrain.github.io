+ 將陣列經過排序後，就變成一個簡單的 sliding window 問題
```cpp
class Solution {
public:
    int maximumBeauty(vector<int>& nums, int k) {
        sort(nums.begin(), nums.end());
        int res = 0, left = 0, right = 0, n = nums.size();
        while (right < n) {
            int num = nums[right++];
            while (num - nums[left] > 2 * k) left++;
            res = max(res, right - left);
        }
        return res;
    }
};
```
