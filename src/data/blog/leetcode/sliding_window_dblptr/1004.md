+ 簡單的不定長 sliding window 問題。
```cpp
class Solution {
public:
    int longestOnes(vector<int>& nums, int k) {
        int res = 0, left = 0, right = 0, n = nums.size();
        int curr = 0;
        int cnt = 0;
        while (right < n) {
            int num = nums[right++];
            while (cnt == k && num == 0) {
                if (nums[left++] == 0) cnt--;
            }
            if (num == 0) cnt++;
            res = max(res, right-left);
        }
        return res;
    }
};
```
