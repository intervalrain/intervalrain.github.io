+ 定長度的 sliding window，秒殺
```cpp
class Solution {
public:
    double findMaxAverage(vector<int>& nums, int k) {
        int sum = 0;
        for (int i = 0; i < k; i++) {
            sum += nums[i];
        }
        int res = sum;
        for (int i = k; i < nums.size(); i++) {
            sum += nums[i];
            sum -= nums[i-k];
            res = max(res, sum);
        }
        return res/(double)k;
    }
};
```
